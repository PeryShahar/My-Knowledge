    • Docker Code Examples
      // Create a file named Dockerfile in the root directory of your React project with the following content:

# Base image
FROM node:14 as build

# Set working directory
WORKDIR /app

# Copy package.json and package-lock.json
COPY package*.json ./

# Install dependencies
RUN npm install

# Copy project files
COPY . .

# Build the React app
RUN npm run build

# Production image
FROM node:14-alpine

# Set working directory
WORKDIR /app

# Copy built React app from the build stage
COPY --from=build /app/build ./build

# Install server dependencies
RUN npm install express

# Copy the server code
COPY server.js ./

# Expose the desired port (change if necessary)
EXPOSE 5000

# Command to run the server
CMD ["node", "server.js"]
