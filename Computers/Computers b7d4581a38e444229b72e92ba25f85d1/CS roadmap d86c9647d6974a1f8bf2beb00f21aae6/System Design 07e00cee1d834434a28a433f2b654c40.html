      // proxy.middleware.ts

import { Injectable, NestMiddleware } from '@nestjs/common';
import { Request, Response } from 'express';
import * as proxy from 'http-proxy-middleware';

@Injectable()
export class ProxyMiddleware implements NestMiddleware {
  private proxyInstance: any;

  constructor() {
    // Configure the target server for the proxy
    this.proxyInstance = proxy({
      target: 'http://example.com', // Replace with the target server's URL
      changeOrigin: true, // Changes the "Host" header to the target URL
    });
  }

  use(req: Request, res: Response, next: () => void) {
    // Call the proxy instance to handle the incoming request
    this.proxyInstance(req, res, next);
  }
}
      // app.module.ts

import { Module, NestModule, MiddlewareConsumer } from '@nestjs/common';
import { ProxyMiddleware } from './proxy.middleware';

@Module({})
export class AppModule implements NestModule {
  configure(consumer: MiddlewareConsumer) {
    // Apply the ProxyMiddleware to all incoming requests
    consumer.apply(ProxyMiddleware).forRoutes('*');
  }
}
