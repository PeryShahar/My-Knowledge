      • Joins / Relations
        // One-to-Many (A child can have multiple toys, but each toy belongs to only one child):
-- Children Table
CREATE TABLE children (
  child_id INT PRIMARY KEY,
  child_name VARCHAR(50),
  favorite_toy_id INT
);

-- Toys Table
CREATE TABLE toys (
  toy_id INT PRIMARY KEY,
  toy_name VARCHAR(50),
  color VARCHAR(20)
);

-- Example of a join for One-to-Many
SELECT child_name, toy_name, color
FROM children
JOIN toys ON children.favorite_toy_id = toys.toy_id;
        // Many-to-One (Multiple children can have the same favorite toy):
-- Children Table (same as before)
CREATE TABLE children (
  child_id INT PRIMARY KEY,
  child_name VARCHAR(50),
  favorite_toy_id INT
);

-- Toys Table (same as before)
CREATE TABLE toys (
  toy_id INT PRIMARY KEY,
  toy_name VARCHAR(50),
  color VARCHAR(20)
);

-- Example of a join for Many-to-One
SELECT child_name, toy_name, color
FROM children
JOIN toys ON children.favorite_toy_id = toys.toy_id;
        // Many-to-Many (Children can have multiple toys, and each toy can be liked by many children):
-- Children Table
CREATE TABLE children (
  child_id INT PRIMARY KEY,
  child_name VARCHAR(50)
);

-- Toys Table
CREATE TABLE toys (
  toy_id INT PRIMARY KEY,
  toy_name VARCHAR(50),
  color VARCHAR(20)
);

-- A junction table to represent the many-to-many relationship
CREATE TABLE children_toys (
  child_id INT,
  toy_id INT,
  PRIMARY KEY (child_id, toy_id),
  FOREIGN KEY (child_id) REFERENCES children (child_id),
  FOREIGN KEY (toy_id) REFERENCES toys (toy_id)
);

-- Example of a join for Many-to-Many
SELECT child_name, toy_name, color
FROM children
JOIN children_toys ON children.child_id = children_toys.child_id
JOIN toys ON children_toys.toy_id = toys.toy_id;
