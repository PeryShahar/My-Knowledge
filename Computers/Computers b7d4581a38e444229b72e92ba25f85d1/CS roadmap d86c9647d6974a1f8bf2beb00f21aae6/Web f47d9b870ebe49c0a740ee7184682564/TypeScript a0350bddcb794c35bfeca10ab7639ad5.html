// Function that never returns or throws an exception
function infiniteLoop(): never {
  while (true) {
    // code that never exits
  }
}

// Variable with type 'never'
let result: never;
result = infiniteLoop(); // Valid assignment

// Unreachable code
function error(message: string): never {
  throw new Error(message);
}

// Type checking using 'never'
function validateNumber(value: number | string): number {
  if (typeof value === 'number') {
    return value;
  } else {
    error('Invalid number');
  }
}

    • Generics
      class Box<T> {
  private item: T;

  addItem(item: T) {
    this.item = item;
  }

  getItem(): T {
    return this.item;
  }
}

const bookBox = new Box<string>(); // Creating a Box that can hold strings
bookBox.addItem("The Great Gatsby");
console.log(bookBox.getItem()); // Output: The Great Gatsby

const toyBox = new Box<number>(); // Creating a Box that can hold numbers
toyBox.addItem(42);
console.log(toyBox.getItem()); // Output: 42
