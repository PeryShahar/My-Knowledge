    • Hoisting
      // Example 1: Variable Hoisting
console.log(name); // Output: undefined
var name = "John";
console.log(name); // Output: John

/* 
In this example, 
the variable name is used before it is declared.
Despite this, the code doesn't throw an error.
This is because during hoisting, the declaration var name; is moved to the top, 
but the assignment (name = "John";) remains in its original place. 
So, the first console.log prints undefined because the variable exists but has not been assigned a value yet.
*/
      // Example 2: Function Hoisting
greet(); // Output: Hello there!
function greet() {
  console.log("Hello there!");
}

/*
In this example,
the function greet() is called before its actual declaration.
The code executes successfully because during hoisting, 
the function declaration is moved to the top. 
Thus, when the greet() function is called, it already exists in memory.
*/

    // Without currying
function add(x, y) {
  return x + y;
}

// With currying
function curryAdd(x) {
  return function (y) {
    return x + y;
  };
}

console.log(curryAdd(3)(5));  // Output: 8
